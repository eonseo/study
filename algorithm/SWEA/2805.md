```cpp
#include <iostream>
#include <string>
using namespace std;

int main() {
	int t, n, result, center;
	string str;

	cin >> t;
	for (int tc = 1; tc <= t; tc++) {
		cin >> n;
		center = n / 2;
		result = 0;
		for (int i = 0; i < center; i++) {
			cin >> str;
			for (int j = center - i; j <= center + i; j++) {
				result += (str[j] - '0');
			}
		}
		for (int i = 0; i <= center; i++) {
			cin >> str;
			for (int j = 0 + i; j < n - i; j++) {
				result += (str[j] - '0');
			}
		}
		cout << "#" << tc << " " << result << "\n";
	}
}
```

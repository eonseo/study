문제 이름 또는 번호: 1486 / 장훈이의 높은 선반

코드(선택): cpp

오늘의 메모: 비트마스킹을 이용한 부분집합으로 풀었다.

초기코드

```cpp
#include <iostream>
#include <algorithm>
#include <limits.h>
using namespace std;
 
int n, b, result, minn;
int height[20];
 
void init() {
	minn = INT_MAX;
}
 
void subset() {
	for (int i = 0; i < (1 << n); i++) {
		result = 0;
		for (int j = 0; j < n; j++) {
			if (i & (1 << j)) result += height[j];
		}
		if (result >= b) minn = min(minn, result);
	}
}
 
int main() {
	ios::sync_with_stdio(false); cin.tie(NULL);
	int t;
 
	cin >> t;
	for (int tc = 1; tc <= t; tc++) {
		init();
		cin >> n >> b;
		for (int i = 0; i < n; i++) {
			cin >> height[i];
		}
		subset();
		cout << "#" << tc << " " << minn - b << "\n";
	}
	return 0;
}
```

문제 이름 또는 번호: 1210 / Ladder

코드(선택): cpp

오늘의 메모: 간단했음

초기코드

```cpp
#include <iostream>
using namespace std;

int map[100][100];
int arriveAt;

int move() {
	int y = 99;
	int x = arriveAt;
	while (y > 0) {
		if (x != 0 && map[y][x - 1] == 1) {
			while (x != 0 && map[y][x - 1] == 1) x--;
		}
		else if (x != 99 && map[y][x + 1] == 1) {
			x += 1;
			while (x != 99 && map[y][x + 1] == 1) x++;
		}
		y--;
	}
	return x;
}

int main()
{
	ios::sync_with_stdio(false);
	cin.tie(NULL);

	for (int i = 0; i < 10; i++) {
		int tc;
		cin >> tc;
		for (int y = 0; y < 100; y++) {
			for (int x = 0; x < 100; x++) {
				cin >> map[y][x];
				if (map[y][x] == 2) {
					arriveAt = x;
				}
			}
		}
		cout << "#" << tc << " " << move() << "\n";
	}
}
```

오늘의 메모(선택): 단순구현

```cpp
#include <iostream>
#include <limits.h>
#include <algorithm>
using namespace std;

int n, m, maximum;
int map[15][15];

void init() {
	memset(map, 0, sizeof(map));
	maximum = -1;
}

void solution(int y, int x) {
	int result = 0;

	for (int i = y; i >= (y - m + 1); i--) {
		for (int j = x; j >= (x - m + 1); j--) {
			result += map[i][j];
		}
	}
	maximum = max(maximum, result);
}

void input() {
	for (int y = 0; y < n; y++) {
		for (int x = 0; x < n; x++) {
			cin >> map[y][x];
			if (y >= (m - 1) && x >= (m - 1)) solution(y, x);
		}
	}
}

int main() {
	ios::sync_with_stdio(false);
	cin.tie(NULL);

	int t;
	cin >> t;
	for (int tc = 1; tc <= t; tc++) {
		cin >> n >> m;
		init();
		input();
		cout << "#" << tc << " " << maximum << "\n";
	}
}
```
